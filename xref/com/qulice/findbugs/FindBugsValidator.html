57  /**
58   * Validates source code and compiled binaris with FindBugs.
59   *
60   * @author Yegor Bugayenko (yegor@tpc2.com)
61   * @version $Id$
62   */
63  public final class FindBugsValidator implements Validator {
64  
65      /**
66       * {@inheritDoc}
67       * @checkstyle RedundantThrows (3 lines)
68       */
69      @Override
70      public void validate(final Environment env) throws ValidationException {
71          if (env.outdir().exists()) {
72              if (!env.exclude("findbugs", "")) {
73                  this.check(this.findbugs(env));
74              }
75          } else {
76              Logger.info(
77                  this,
78                  "No classes at %s, no FindBugs validation",
79                  env.outdir()
80              );
81          }
82      }
83  
84      /**
85       * Start findbugs and return its output.
86       * @param env Environment
87       * @return Output of findbugs
88       */
89      private String findbugs(final Environment env) {
90          final List<String> args = new LinkedList<String>();
91          args.add("java");
92          args.addAll(this.options(env));
93          args.add(Wrap.class.getName());
94          args.add(env.basedir().getPath());
95          args.add(env.outdir().getPath());
96          // @checkstyle MultipleStringLiteralsCheck (2 lines)
97          args.add(StringUtils.join(env.classpath(), ",")
98              .replace("\\", "/")
99          );
100         return new VerboseProcess(new ProcessBuilder(args)).stdoutQuietly();
101     }
102 
103     /**
104      * Java options.
105      * @param env Environment
106      * @return Options
107      */
108     private Collection<String> options(final Environment env) {
109         final File jar = this.jar(Wrap.class);
110         final Collection<String> opts = new LinkedList<String>();
111         opts.add("-classpath");
112         opts.add(
113             StringUtils.join(
114                 CollectionUtils.union(
115                     Arrays.asList(
116                         jar,
117                         this.jar(FindBugs2.class),
118                         this.jar(ClassFormatException.class),
119                         this.jar(DocumentException.class),
120                         this.jar(JaxenException.class),
121                         this.jar(ClassNode.class),
122                         this.jar(ClassVisitor.class),
123                         this.jar(When.class),
124                         this.jar(EmptyVisitor.class),
125                         this.jar(FormatterNumberFormatException.class)
126                     ),
127                     env.classpath()
128                 ),
129                 System.getProperty("path.separator")
130             ).replace("\\", "/")
131         );
132         return opts;
133     }
134 
135     /**
136      * Get address of our JAR or directory.
137      * @param resource Name of resource
138      * @return The file
139      */
140     private File jar(final Class<?> resource) {
141         final String name = resource.getName()
142             // @checkstyle MultipleStringLiteralsCheck (1 line)
143             .replace(".", "/");
144         final URL res = this.getClass().getResource(
145             String.format("/%s.class", name)
146         );
147         if (res == null) {
148             throw new IllegalStateException(
149                 String.format(
150                     "can't find JAR for %s",
151                     name
152                 )
153             );
154         }
155         final String path = res.getFile().replaceAll("\\!.*$", "");
156         File file;
157         if ("jar".equals(FilenameUtils.getExtension(path))) {
158             file = new File(URI.create(path).getPath());
159         } else {
160             file = new File(path).getParentFile()
161                 .getParentFile()
162                 .getParentFile()
163                 .getParentFile();
164         }
165         Logger.debug(this, "#jar(%s): found at %s", resource.getName(), file);
166         return file;
167     }

169     /**
170      * Check report for errors.
171      * @param report The report
172      * @throws ValidationException If it contains errors
173      * @checkstyle RedundantThrows (3 lines)
174      */
175     private void check(final String report) throws ValidationException {
176         int total = 0;
177         for (final String line
178             : report.split(System.getProperty("line.separator"))) {
179             if (line.matches("[a-zA-Z ]+: .*")) {
180                 Logger.warn(this, "FindBugs: %s", line);
181                 ++total;
182             }
183         }
184         if (total > 0) {
185             throw new ValidationException(
186                 "%d FindBugs violations (see log above)",
187                 total
188             );
189         }
190     }
