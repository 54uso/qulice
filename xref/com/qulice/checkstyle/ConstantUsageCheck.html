30  package com.qulice.checkstyle;
31  
32  import com.puppycrawl.tools.checkstyle.api.Check;
33  import com.puppycrawl.tools.checkstyle.api.DetailAST;
34  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
35  
36  /**
37   * Checks that constant, declared as private field of class is used more than
38   * once.
39   *
40   * <p>We're still experimenting with this check and it is not used in
41   * released version.
42   *
43   * @author Dmitry Bashkin (dmitry.bashkin@qulice.com)
44   * @version $Id$
45   */
46  public final class ConstantUsageCheck extends Check {
47  
48      @Override
49      public int[] getDefaultTokens() {
50          return new int[]{
51              TokenTypes.VARIABLE_DEF,
52          };
53      }
54  
55      /**
56       * {@inheritDoc}
57       * @checkstyle NestedIfDepth (35 lines)
58       */
59      @Override
60      public void visitToken(final DetailAST ast) {
61          if (this.isField(ast) && this.isFinal(ast)) {
62              final DetailAST nameNode = ast.findFirstToken(TokenTypes.IDENT);
63              final String name = nameNode.getText();
64              final int line = nameNode.getLineNo();
65              DetailAST variable = ast.getNextSibling();
66              int counter = 0;
67              while (null != variable) {
68                  if (TokenTypes.VARIABLE_DEF == variable.getType()) {
69                      final DetailAST assign =
70                          variable.findFirstToken(TokenTypes.ASSIGN);
71                      final DetailAST expression =
72                          assign.findFirstToken(TokenTypes.EXPR);
73                      final String text = this.getText(expression);
74                      if (text.contains(name)) {
75                          counter = counter + 1;
76                      }
77                  } else {
78                      counter = counter + this.parseMethod(variable, name);
79                  }
80                  variable = variable.getNextSibling();
81              }
82              if (counter < 2) {
83                  this.log(
84                      line,
85                      String.format("Constant \"%s\" used only once", name)
86                  );
87              }
88          }
89      }
90  
91      /**
92       * Returns text representation of the specified node, including it's
93       * children.
94       * @param node Node, containing text.
95       * @return Text representation of the node.
96       */
97      private String getText(final DetailAST node) {
98          String ret;
99          if (0 == node.getChildCount()) {
100             ret = node.getText();
101         } else {
102             final StringBuilder result = new StringBuilder();
103             DetailAST child = node.getFirstChild();
104             while (null != child) {
105                 final String text = this.getText(child);
106                 result.append(text);
107                 child = child.getNextSibling();
108             }
109             ret = result.toString();
110         }
111         return ret;
112     }
113 
114     /**
115      * Returns <code>true</code> if specified node has parent node of type
116      * <code>OBJBLOCK</code>.
117      * @param node Node to check.
118      * @return True if parent node is <code>OBJBLOCK</code>, else
119      *  returns <code>false</code>.
120      */
121     private boolean isField(final DetailAST node) {
122         final DetailAST parent = node.getParent();
123         return TokenTypes.OBJBLOCK == parent.getType();
124     }
125 
126     /**
127      * Returns true if specified node has modifiers of type <code>FINAL</code>.
128      * @param node Node to check.
129      * @return True if specified node contains modifiers of type
130      *  <code>FINAL</code>, else returns <code>false</code>.
131      */
132     private boolean isFinal(final DetailAST node) {
133         final DetailAST modifiers = node.findFirstToken(TokenTypes.MODIFIERS);
134         return modifiers.branchContains(TokenTypes.FINAL);
135     }
136 
137     /**
138      * Parses the body of the method and increments counter each time when it
139      * founds constant name.
140      * @param method Tree node, containing method.
141      * @param name Constant name to search.
142      * @return Number of found constant usages.
143      */
144     private int parseMethod(final DetailAST method, final String name) {
145         int counter = 0;
146         final DetailAST opening = method.findFirstToken(TokenTypes.SLIST);
147         if (null != opening) {
148             final DetailAST closing = opening.findFirstToken(TokenTypes.RCURLY);
149             final int start = opening.getLineNo();
150             final int end = closing.getLineNo() - 1;
151             final String[] lines = this.getLines();
152             for (int pos = start; pos < end; pos += 1) {
153                 if (lines[pos].contains(name)) {
154                     counter += 1;
155                 }
156             }
157         }
158         return counter;
159     }
